  likeBtn.addEventListener('click', function() {
    // Получаем текущее состояние лайка
    const isLiked = likeBtn.classList.contains('-active-like');
  
    // Обновляем состояние лайка
    likeBtn.classList.toggle('-active-like');
    comments[0].isLiked =!isLiked;
  
    // Обновляем текущее число лайков
    likesCount.textContent = comments[0].likesCount + (isLiked? -1 : 1);
  });
